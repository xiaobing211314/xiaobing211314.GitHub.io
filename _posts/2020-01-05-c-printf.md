### C语言三种打印方式

1.
```c
#include <stdio.h>
int main()
{
	    printf("\n\
         @\n\
        / \\\n\
        * *\n\
        * *\n\
        * *\n\
    * * * * * *\n\
  * * * * * * * *\n\
* * * * * * * * * *\n\
        * *\n\
        * *\n\
      * * * *\n\
    * * * * * *\n"); 
	return 0;
}
```

2.
```c
  #include <stdio.h>
  
  int main()
  {

          printf("    * * * * * *\n");
          printf("  * * * * * * * *\n");
          printf("* * * * * * * * * *\n");
          printf("        * *\n");
          printf("        * *\n");
          printf("      * * * *\n");
          printf("    * * * * * *\n");
         return 0;
  
  }
```
